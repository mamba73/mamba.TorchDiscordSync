/// <summary>
/// Find a Discord user by their username in the configured guild
/// </summary>
private SocketUser FindUserByUsername(string username)
{
    try
    {
        if (_client == null || !_isReady)
        {
            LoggerUtil.LogError("[DISCORD_BOT] Client not ready for user search");
            return null;
        }

        var guild = _client.GetGuild(_config.GuildID);
        if (guild == null)
        {
            LoggerUtil.LogError($"[DISCORD_BOT] Guild not found: {_config.GuildID}");
            return null;
        }

        // Search in guild members
        var user = guild.Users.FirstOrDefault(u => 
            u.Username.Equals(username, StringComparison.OrdinalIgnoreCase) ||
            u.GlobalName?.Equals(username, StringComparison.OrdinalIgnoreCase) == true
        );

        if (user == null)
        {
            LoggerUtil.LogWarning($"[DISCORD_BOT] User not found in guild: {username}");
            LoggerUtil.LogDebug($"[DISCORD_BOT] Guild has {guild.Users.Count} members");
        }
        else
        {
            LoggerUtil.LogInfo($"[DISCORD_BOT] Found user: {user.Username} ({user.Id})");
        }

        return user;
    }
    catch (Exception ex)
    {
        LoggerUtil.LogError($"[DISCORD_BOT] FindUserByUsername error: {ex.Message}");
        return null;
    }
}
```

/// <summary>
/// Automatic reconnect with exponential backoff
/// </summary>
private async Task OnBotDisconnected(Exception ex)
{
    LoggerUtil.LogError($"[DISCORD_BOT] Disconnected: {ex?.Message}");
    _isReady = false;
    _isConnected = false;

    // Automatic reconnect after 5 seconds
    await Task.Delay(5000);
    LoggerUtil.LogWarning("[DISCORD_BOT] Attempting reconnect...");
    
    int retryCount = 0;
    int maxRetries = 5;
    int retryDelay = 5000; // Start with 5 seconds

    while (retryCount < maxRetries && !_isConnected)
    {
        try
        {
            retryCount++;
            LoggerUtil.LogInfo($"[DISCORD_BOT] Reconnect attempt {retryCount}/{maxRetries}");
            
            await ReconnectAsync();
            
            if (_isConnected)
            {
                LoggerUtil.LogSuccess("[DISCORD_BOT] Reconnected successfully");
                return;
            }
        }
        catch (Exception reconnectEx)
        {
            LoggerUtil.LogError($"[DISCORD_BOT] Reconnect attempt {retryCount} failed: {reconnectEx.Message}");
        }

        // Exponential backoff
        retryDelay *= 2;
        if (retryDelay > 60000) retryDelay = 60000; // Max 1 minute
        
        await Task.Delay(retryDelay);
    }

    if (!_isConnected)
    {
        LoggerUtil.LogError($"[DISCORD_BOT] Failed to reconnect after {maxRetries} attempts");
    }
}

/// <summary>
/// Reconnect the bot
/// </summary>
public async Task<bool> ReconnectAsync()
{
    try
    {
        if (_client != null)
        {
            await _client.StartAsync();
            _isConnected = true;
            LoggerUtil.LogSuccess("[DISCORD_BOT] Reconnect successful");
            return true;
        }
        return false;
    }
    catch (Exception ex)
    {
        LoggerUtil.LogError($"[DISCORD_BOT] Reconnect failed: {ex.Message}");
        return false;
    }
}
```
