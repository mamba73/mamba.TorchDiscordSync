```csharp
private async Task OnBotDisconnected(Exception ex)
{
    LoggerUtil.LogError($"[DISCORD_BOT] Disconnected: {ex?.Message}");
    _isReady = false;
    _isConnected = false;
    
    // Automatic reconnect after 5 seconds
    await Task.Delay(5000);
    LoggerUtil.LogWarning("[DISCORD_BOT] Attempting reconnect...");
    await ReconnectAsync();
}

public async Task ReconnectAsync()
{
    try
    {
        if (_client != null)
        {
            await _client.StartAsync();
            LoggerUtil.LogSuccess("[DISCORD_BOT] Reconnected successfully");
        }
    }
    catch (Exception ex)
    {
        LoggerUtil.LogError($"[DISCORD_BOT] Reconnect failed: {ex.Message}");
    }
}
```
